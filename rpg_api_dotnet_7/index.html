    데이터베이스 컨텍스트 (DatabaseContext)¶

    Microsoft.EntityFrameworkCore Microsoft.EntityFrameworkCore.Tools Microsoft.EntityFrameworkCore.InMemory

    RpgServer/RpgApi/UserDbContext.cs

    using Microsoft.Data.Sqlite;
using Microsoft.EntityFrameworkCore;
using Rpg.Models;

namespace Rpg
{
    public class UserDbContext : DbContext
    {
        public UserDbContext(DbContextOptions<UserDbContext> opts) : base(opts)
        {
        }

        public async Task<Player?> FindPlayer(int id)
        {
            return await PlayerSet.FindAsync(id);
        }

        public async Task<Player> MakePlayer()
        {
            Player newPlayer = new Player();
            Add(newPlayer);
            await SaveChangesAsync();
            return newPlayer;
        }

        public DbSet<Player> PlayerSet { get; set; }
    }
}

    RpgServer/RpgApi/Program.cs

    var builder = WebApplication.CreateBuilder(args);
builder.Services.AddDbContext<UserDbContext>(opts => opts.UseInMemoryDatabase("UserDb"));

var app = builder.Build();
app.MapGet("/player/{id}", async (int id, UserDbContext ctx) => await ctx.FindPlayer(id));
app.MapPost("/player", async (UserDbContext ctx) => await ctx.MakePlayer());
